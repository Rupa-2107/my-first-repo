# Student Portal Application with Basic Student Info

students = {}

def register_student():
    print("\n--- Student Registration ---")
    suc_no = input("Enter SUC Number (Student ID): ")
    
    if suc_no in students:
        print("SUC Number already exists! Please log in.")
        return

    name = input("Enter Full Name: ")
    course = input("Enter Course: ")
    college = input("Enter College Name: ")
    email = input("Enter Email: ")
    phone = input("Enter Phone Number: ")

    students[suc_no] = {
        'name': name,
        'course': course,
        'college': college,
        'email': email,
        'phone': phone
    }

    print("Student registered successfully!")
